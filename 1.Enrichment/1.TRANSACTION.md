Formato da notificação de atualização de transação enviado pelo sistema.

```ts
type TransactionEvent = {
	event: "transaction",
	payload: {
		contractDisputeId: string,
		transactionIdentifier: string,
		acquirerReferenceNumber: string,
	}
}
```

```ts
type TransactionResponse = {
	transactionIdentifier: string,
	authorizationDate: string,
	authorizationCurrencyCode: string, // ISO 4217
	acquirerReferenceNumber: string,
	authorizationSequentialNumber: string,
	authorizationCodeResponse: string,
	authorizationCode: string,
	cardNumber: string,
	terminalIdentifier: string,
	authorizedAmount: number,
	paymentPlan: number,
	isInternational: boolean,
	
	// Associations
	// tip: Search the values in our API
	authorizationCardDataMethodId: string,
	authorizationCnpSecurityLevelId: string,

	installments: [
		{
			acquirerReferenceNumber: string,
			installmentNumber: number,
			installmentDate: string,
			amount: number,
			issuerSettlementStatus: "settled" | "scheduled" | "canceled",
			merchantSettlementStatus: "settled" | "scheduled" | "canceled",
		}
	],
}
```
